return_all_pair=function(x){
  tmp = x  %>% gsub(", \\.\\.\\.", "", .) %>% strsplit(", ")  %>% unlist()
  if(length(tmp)>1){
    tmp = t(combn(tmp, 2))
  }
  return(tmp)
}
